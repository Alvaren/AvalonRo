-	script	PvP Coordinator	-1,{
	OnPCLogoutEvent:
		if(hidden > 0){
			set .@players$, strcharinfo(0);
			set .@player, getcharid(3);
			detachrid;
			atcommand "@unmute "+ .@players$;
			attachrid(.@player);
			atcommand "@hide";
			atcommand "@costume";
		}
		end;

	OnPCLoginEvent:
		getmapxy (.@map$, .@x, .@y, 0);
		if( .@map$ == "quiz_01" || .@map$ == "quiz_02" || .@map$ == "06guild_01"){
			if(hidden > 0){
				atcommand "@costume Xmas";
				atcommand "@hide";
				atcommand "@mute 3000 " + strcharinfo(0);
				detachrid;
			}
		}
		end;
}
